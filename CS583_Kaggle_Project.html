def normalize_data(train, test):
    '''
        Normalizing the data using the Mean - Standard Deviation Method
        x = x - mean / std
    '''
    mean = train.iloc[:,2:].mean(axis=0)
    std = train.iloc[:,2:].std(axis=0)
    
    train = train.iloc[:,2:] - mean
    train = train.iloc[:,2:] / std
    
    return train
